CREATE TABLE PETS(
   ID   INT              NOT NULL,
   NAME VARCHAR (20),
   WEIGHT VARCHAR(20),
   AGE    INT  ,
  TYPE    INT ,
  OWNER_ID    INT       NOT NULL
   CREATE_DATE   DATE,       
   PRIMARY KEY (ID,OWNER_ID)
);


CREATE TABLE OWNERS(
   ID   INT              NOT NULL,
   NAME VARCHAR (20)    NOT NULL,
   ADDRESS VARCHAR(20)  NOT NULL,
   CITY    VARCHAR(20) ,
  STATE    VARCHAR(20)     NOT NULL ,
  ZIP   INT       NOT NULL
   COUNTRY    VARCHAR(20)    NOT NULL,       
   PRIMARY KEY (ID)
);
